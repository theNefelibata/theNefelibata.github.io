    class Solution:
	def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
		queue = []
		if k == 0:
			return False
		for i in nums:
			if i not in queue:
				if len(queue) < k:
					queue.append(i)
				else:
					queue.pop(0)
					queue.append(i)
			else:
				return True
		return False

    class Solution:
	def containsNearbyAlmostDuplicate(self, nums: List[int], k: int, t: int) -> bool:
		def getID(x, t):
			return (x+1)//(t+1)-1 if x<0 else x//(t+1)
		
		n = len(nums)
		m = {}
		for i in range(n):
			x = nums[i]
			idx = getID(x, t)
			if idx in m:
				return True
			l , r = idx-1, idx+1
			if l in m and abs(x - m[l]) <= t:
				return True
			if r in m and abs(x - m[r]) <= t:
				return True
			m[idx] = x
			if i >= k:
				m.pop(getID(nums[i-k],t))
		return False
